    <script>

        // Application State

        let currentDate = new Date();

        let tasks = JSON.parse(localStorage.getItem('focusPlannerTasks') || '{}');

        let distractions = JSON.parse(localStorage.getItem('focusPlannerDistractions') || '[]');

        let stats = JSON.parse(localStorage.getItem('focusPlannerStats') || '{"tasksCompleted": 0, "focusTime": 0, "distractionsCaught": 0}');

        let selectedPriority = 'medium';

        let timerInterval = null;

        let timerSeconds = 25 * 60; // 25 minutes default

        let isTimerRunning = false;

        let breathingInterval = null;

 

        // Focus tips for motivation

        const focusTips = [

            "Break large tasks into 25-minute focused sessions (Pomodoro technique)",

            "Write down distractions instead of acting on them immediately",

            "Use the 2-minute rule: if it takes less than 2 minutes, do it now",

            "Create a 'parking lot' for ideas that pop up during focused work",

            "Schedule specific times for checking messages and social media",

            "Use background noise or music to help maintain focus",

            "Take regular breaks to prevent mental fatigue",

            "Celebrate small wins to maintain motivation",

            "Start with your most challenging task when your energy is highest",

            "Use visual cues to remind yourself to stay on task"

        ];

 

        // Initialize the application

        function initApp() {

            generateTimeGrid();

            updateCalendarHeader();

            renderTasks();

            renderDistractions();

            updateStats();

            showNewTip();

            setupPriorityButtons();

           

            // Set current time as default for new tasks

            const now = new Date();

            document.getElementById('taskTime').value = now.getHours().toString().padStart(2, '0') + ':' + now.getMinutes().toString().padStart(2, '0');

        }

 

        // Generate the time grid for the calendar

        function generateTimeGrid() {

            const grid = document.getElementById('timeGrid');

            grid.innerHTML = '';

 

            for (let hour = 6; hour < 23; hour++) {

                // Time label

                const timeLabel = document.createElement('div');

                timeLabel.className = 'time-label';

                timeLabel.textContent = `${hour.toString().padStart(2, '0')}:00`;

                grid.appendChild(timeLabel);

 

                // Time slot

                const timeSlot = document.createElement('div');

                timeSlot.className = 'time-slot';

                timeSlot.dataset.hour = hour;

                timeSlot.onclick = () => quickSchedule(hour);

                grid.appendChild(timeSlot);

            }

        }

 

        // Setup priority button interactions

        function setupPriorityButtons() {

            document.querySelectorAll('.priority-btn').forEach(btn => {

                btn.onclick = () => {

                    document.querySelectorAll('.priority-btn').forEach(b => b.classList.remove('active'));

                    btn.classList.add('active');

                    selectedPriority = btn.dataset.priority;

                };

            });

        }

 

        // Quick schedule by clicking on time slot

        function quickSchedule(hour) {

            const taskInput = document.getElementById('taskInput');

            if (taskInput.value.trim()) {

                const time = `${hour.toString().padStart(2, '0')}:00`;

                document.getElementById('taskTime').value = time;

                addTask();

            } else {

                taskInput.focus();

                taskInput.placeholder = `What will you do at ${hour}:00?`;

                setTimeout(() => {

                    taskInput.placeholder = "What needs your attention?";

                }, 3000);

            }

        }

 

        // Add a new task

        function addTask() {

            const taskText = document.getElementById('taskInput').value.trim();

            const taskTime = document.getElementById('taskTime').value;

            const taskDuration = parseInt(document.getElementById('taskDuration').value) || 60;

 

            if (!taskText) {

                alert('Please enter a task description!');

                return;

            }

 

            const dateKey = currentDate.toDateString();

            if (!tasks[dateKey]) {

                tasks[dateKey] = [];

            }

 

            const newTask = {

                id: Date.now(),

                text: taskText,

                time: taskTime,

                duration: taskDuration,

                priority: selectedPriority,

                completed: false

            };

 

            tasks[dateKey].push(newTask);

            localStorage.setItem('focusPlannerTasks', JSON.stringify(tasks));

 

            // Clear inputs

            document.getElementById('taskInput').value = '';

            document.getElementById('taskDuration').value = '60';

 

            renderTasks();

            showCelebration('📅 Task scheduled!');

        }

 

        // Render tasks in the calendar

        function renderTasks() {

            // Clear existing task blocks

            document.querySelectorAll('.task-block').forEach(block => block.remove());

 

            const dateKey = currentDate.toDateString();

            const todayTasks = tasks[dateKey] || [];

 

            todayTasks.forEach(task => {

                const [hour, minute] = task.time.split(':').map(Number);

                const slot = document.querySelector(`[data-hour="${hour}"]`);

               

                if (slot) {

                    const taskBlock = document.createElement('div');

                    taskBlock.className = `task-block ${task.priority}-priority`;

                    if (task.completed) {

                        taskBlock.style.opacity = '0.6';

                        taskBlock.style.textDecoration = 'line-through';

                    }

                   

                    taskBlock.innerHTML = `

                        <strong>${task.time}</strong> (${task.duration}min)<br>

                        ${task.text}

                        <div style="position: absolute; top: 5px; right: 8px; cursor: pointer;" onclick="toggleTask(${task.id})">

                            ${task.completed ? '✅' : '⏱️'}

                        </div>

                    `;

                   

                    taskBlock.onclick = (e) => {

                        e.stopPropagation();

                        if (e.target.onclick !== toggleTask) {

                            editTask(task.id);

                        }

                    };

                   

                    slot.appendChild(taskBlock);

                }

            });

        }

 

        // Toggle task completion

        function toggleTask(taskId) {

            const dateKey = currentDate.toDateString();

            const task = tasks[dateKey].find(t => t.id === taskId);

           

            if (task) {

                task.completed = !task.completed;

                localStorage.setItem('focusPlannerTasks', JSON.stringify(tasks));

               

                if (task.completed) {

                    stats.tasksCompleted++;

                    stats.focusTime += task.duration;

                    localStorage.setItem('focusPlannerStats', JSON.stringify(stats));

                    showCelebration('🎉 Task completed!');

                } else {

                    stats.tasksCompleted = Math.max(0, stats.tasksCompleted - 1);

                    stats.focusTime = Math.max(0, stats.focusTime - task.duration);

                    localStorage.setItem('focusPlannerStats', JSON.stringify(stats));

                }

               

                renderTasks();

                updateStats();

            }

        }

 

        // Edit task (simple version - delete and recreate)

        function editTask(taskId) {

            const dateKey = currentDate.toDateString();

            const taskIndex = tasks[dateKey].findIndex(t => t.id === taskId);

           

            if (taskIndex !== -1) {

                const task = tasks[dateKey][taskIndex];

                const newText = prompt('Edit task:', task.text);

               

                if (newText && newText.trim()) {

                    task.text = newText.trim();

                    localStorage.setItem('focusPlannerTasks', JSON.stringify(tasks));

                    renderTasks();

                } else if (newText === '') {

                    // Delete task if empty

                    tasks[dateKey].splice(taskIndex, 1);

                    localStorage.setItem('focusPlannerTasks', JSON.stringify(tasks));

                    renderTasks();

                }

            }

        }

 

        // Capture distraction

        function captureDistraction() {

            const distractionText = document.getElementById('distractionInput').value.trim();

           

            if (!distractionText) return;

 

            const distraction = {

                id: Date.now(),

                text: distractionText,

                timestamp: new Date().toLocaleTimeString(),

                date: new Date().toDateString()

            };

 

            distractions.unshift(distraction);

            localStorage.setItem('focusPlannerDistractions', JSON.stringify(distractions));

           

            stats.distractionsCaught++;

            localStorage.setItem('focusPlannerStats', JSON.stringify(stats));

 

            document.getElementById('distractionInput').value = '';

            renderDistractions();

            updateStats();

            showCelebration('🧠 Distraction captured! Back to focus!');

        }

 

        // Render distractions list

        function renderDistractions() {

            const list = document.getElementById('distractionList');

            const todayDistractions = distractions.filter(d => d.date === new Date().toDateString());

           

            list.innerHTML = todayDistractions.slice(0, 5).map(distraction => `

                <div class="distraction-item">

                    <div>

                        <strong>${distraction.timestamp}</strong><br>

                        ${distraction.text}

                    </div>

                    <button class="btn btn-danger" style="padding: 5px 10px; font-size: 0.8em;" onclick="removeDistraction(${distraction.id})">×</button>

                </div>

            `).join('');

        }

 

        // Remove distraction

        function removeDistraction(id) {

            distractions = distractions.filter(d => d.id !== id);

            localStorage.setItem('focusPlannerDistractions', JSON.stringify(distractions));

            renderDistractions();

        }

 

        // Update statistics display

        function updateStats() {

            document.getElementById('tasksCompleted').textContent = stats.tasksCompleted;

            document.getElementById('focusTime').textContent = stats.focusTime;

            document.getElementById('distractionsCaught').textContent = stats.distractionsCaught;

        }

 

        // Show new focus tip

        function showNewTip() {

            const tip = focusTips[Math.floor(Math.random() * focusTips.length)];

        // Pomodoro Timer Functions

        function startTimer() {

            if (!isTimerRunning) {

                isTimerRunning = true;

                timerInterval = setInterval(() => {

                    timerSeconds--;

                    updateTimerDisplay();

                   

                    if (timerSeconds <= 0) {

                        clearInterval(timerInterval);

                        isTimerRunning = false;

                        showCelebration('🍅 Pomodoro completed! Take a break!');

                        timerSeconds = 25 * 60; // Reset to 25 minutes

                        updateTimerDisplay();

                    }

                }, 1000);

            }

        }

 

        function pauseTimer() {

            if (isTimerRunning) {

                clearInterval(timerInterval);

                isTimerRunning = false;

            }

        }

 

        function resetTimer() {

            clearInterval(timerInterval);

            isTimerRunning = false;

            timerSeconds = 25 * 60;

            updateTimerDisplay();

        }

 

        function updateTimerDisplay() {

            const minutes = Math.floor(timerSeconds / 60);

            const seconds = timerSeconds % 60;

            document.getElementById('timerDisplay').textContent =

                `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

        }

        function showBreathingExercise() {

            document.getElementById('breathingModal').style.display = 'block';

        }

 

        function closeBreathingExercise() {

            document.getElementById('breathingModal').style.display = 'none';

            if (breathingInterval) {

                clearInterval(breathingInterval);

            }

        }

 

        function startBreathing() {

            const circle = document.getElementById('breathCircle');

            const instruction = document.getElementById('breathInstruction');

            const btn = document.getElementById('breathBtn');

           

            btn.textContent = 'Stop';

            btn.onclick = stopBreathing;

           

            let phase = 'inhale'; // inhale, hold, exhale, hold

            let count = 0;

           

            breathingInterval = setInterval(() => {

                count++;

               

                switch(phase) {

                    case 'inhale':

                        circle.className = 'breath-circle inhale';

                        instruction.textContent = 'Hold...';

                        if (count >= 2) { phase = 'inhale'; count = 0; }

                        break;

                }

            }, 1000);

        }

 

        function stopBreathing() {

            clearInterval(breathingInterval);

            const circle = document.getElementById('breathCircle');

            const instruction = document.getElementById('breathInstruction');

            const btn = document.getElementById('breathBtn');

           

            circle.className = 'breath-circle';

            instruction.textContent = 'Great job! Feeling more focused?';

            btn.textContent = 'Start Breathing';

            btn.onclick = startBreathing;

           

            setTimeout(() => {

                closeBreathingExercise();

            }, 2000);

        }

 

        // Export calendar data

        function exportCalendar() {

            const dateKey = currentDate.toDateString();

            const todayTasks = tasks[dateKey] || [];

           

            let exportText = `Focus Planner - ${dateKey}\n`;

            exportText += '='.repeat(50) + '\n\n';

           

            exportText += 'SCHEDULED TASKS:\n';

            exportText += '-'.repeat(20) + '\n';

           

            todayTasks.sort((a, b) => a.time.localeCompare(b.time)).forEach(task => {

                const status = task.completed ? '✅ DONE' : '⏱️ PENDING';

                exportText += `${task.time} - ${task.text} (${task.duration}min) [${task.priority.toUpperCase()}] ${status}\n`;

            });

           

            exportText += '\nDISTRACTIONS CAPTURED:\n';

            exportText += '-'.repeat(25) + '\n';

           

            const todayDistractions = distractions.filter(d => d.date === dateKey);

            todayDistractions.forEach(distraction => {

                exportText += `${distraction.timestamp} - ${distraction.text}\n`;

            });

           

            exportText += '\nSTATISTICS:\n';

            exportText += '-'.repeat(15) + '\n';

            exportText += `Tasks Completed: ${stats.tasksCompleted}\n`;

            exportText += `Focus Time: ${stats.focusTime} minutes\n`;

            exportText += `Distractions Caught: ${stats.distractionsCaught}\n`;

           

            // Create downloadable file

            const blob = new Blob([exportText], { type: 'text/plain' });

            const url = window.URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = `focus-planner-${currentDate.toISOString().split('T')[0]}.txt`;

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            window.URL.revokeObjectURL(url);

           

            showCelebration('📁 Schedule exported!');

        }

        function updateTimeDefaults() {

            const now = new Date();

            const currentHour = now.getHours();

            const currentMinute = now.getMinutes();

           

            // Round up to next 15-minute interval

            let suggestedMinute = Math.ceil(currentMinute / 15) * 15;

            let suggestedHour = currentHour;

           

            if (suggestedMinute >= 60) {

                suggestedMinute = 0;

                suggestedHour = (suggestedHour + 1) % 24;

            }

           

            const timeString = `${suggestedHour.toString().padStart(2, '0')}:${suggestedMinute.toString().padStart(2, '0')}`;

            document.getElementById('taskTime').value = timeString;

        }

        setInterval(updateTimeDefaults, 60000);
